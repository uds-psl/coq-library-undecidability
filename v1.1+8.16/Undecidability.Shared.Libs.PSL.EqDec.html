Lemma Dec_reflect (X: Prop) (d: dec X) :
  Dec X <-> X.
Proof.
  destruct d as [A|A]; cbv; firstorder congruence.
Qed.

Tactic Notation "decide" constr(p) :=
  destruct (Dec p).

  destruct (Dec _).
